while (true) {
    if (currentActivity() == "") {
        log("已经在花椒直播间，哦了");
        break;
    } else {
        log("没有在花椒直播间,请看看怎么办");
        //这里可以干很多事，比如：sleep多久 
        //还没进入画面，手机卡住了。再处理等等。 
        sleep(1000)
        exit();
    };
}
